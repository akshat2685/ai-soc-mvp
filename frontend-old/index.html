Quick Integration Snippet (Node.js SDK)

Paste this code directly into your Node.js/Express backend to protect your endpoints with one line of code.

// 1. Save this helper as 'shield_soc.js'
const axios = require('axios');

const SOC_URL = 'http://YOUR_SOC_IP:8000'; // Update with your SOC URL
const API_KEY = 'YOUR_API_KEY_HERE';

async function protectRequest(req) {
    const payload = {
        event_type: req.path.includes('login') ? 'login' : 'api_call',
        source_ip: req.headers['x-forwarded-for'] || req.socket.remoteAddress || '127.0.0.1',
        user_id: req.body?.username || req.body?.email || req.user?.id || null,
        status: 'pending',
        device_id: req.cookies?.device_id || 'unknown',
        user_agent: req.headers['user-agent'] || 'unknown',
        endpoint: req.path,
        method: req.method,
        headers: req.headers
    };

    try {
        const res = await axios.post(`${SOC_URL}/ingest`, payload, {
            headers: { 'X-API-Key': API_KEY },
            timeout: 800 // Fail open quickly if SOC is slow
        });
        return { action: 'ALLOW' };
    } catch (err) {
        if (err.response) {
            if (err.response.status === 401) console.error('SOC API Key is invalid');
            if (err.response.status === 403) return { action: 'BLOCK', detail: 'Suspicious IP' };
            if (err.response.status === 429) return { action: 'RATE_LIMIT', detail: 'Rate limit exceeded' };
        }
        return { action: 'ALLOW' }; // Fail open
    }
}

Python SDK Integration Snippet

# 1. Save this helper as 'shield_soc.py'
import httpx

SOC_URL = 'http://YOUR_SOC_IP:8000' # Update with your SOC URL
API_KEY = 'YOUR_API_KEY_HERE'

def protect_request(ip: str, path: str, method: str, user_agent: str = None, headers: dict = None, user_id: str = None) -> str:
    payload = {
        "event_type": "login" if "login" in path else "api_call",
        "source_ip": ip,
        "user_id": user_id,
        "status": "pending",
        "device_id": "unknown",
        "user_agent": user_agent or "unknown",
        "endpoint": path,
        "method": method,
        "headers": headers or {}
    }
    try:
        resp = httpx.post(f"{SOC_URL}/ingest", json=payload, headers={"X-API-Key": API_KEY}, timeout=0.8)
        if resp.status_code == 403:
            return "BLOCK"
        elif resp.status_code == 429:
            return "RATE_LIMIT"
    except Exception:
        pass
    return "ALLOW" # Fail open
